Хід роботи

Обгрунтування вибору мови програмування

Було обрано мову програмування Scheme. Scheme є функціональною мовою програмування, яка підтримує функції вищого порядку, замикання, рекурсію та інші функціональні концепції. Це робить її потужним інструментом для вирішення складних завдань, особливо в областях, де функціональне програмування є важливим, наприклад, в обробці даних. Вона відома своєю простотою та мінімалізмом в синтаксисі. Вона має дуже невелику кількість ключових слів і структурних елементів мови. Це дозволяє легко розуміти і змінювати код, що робить її відмінним вибором для навчання програмуванню та швидкого прототипування.

(define (find-common-minimum vec1 vec2)
  (define common-elements
    (filter (lambda (x) (memv x vec2)) vec1))

  (if (not (null? common-elements))
      (let* ((min-element (apply min common-elements))
             (min-index (index-of min-element vec1)))
        (display "Перший вектор: ")
        (display vec1)
        (newline)
        (display "Другий вектор: ")
        (display vec2)
        (newline)
        (display "Найменший спільний елемент: ")
        (display min-element)
        (newline)
        (display "Його індекс в першому векторі: ")
        (display min-index)
        (newline))
      (display "Немає спільних елементів.")))

(define (index-of element lst)
  (let loop ((lst lst) (index 0))
    (cond ((null? lst) #f)
          ((equal? element (car lst)) index)
          (else (loop (cdr lst) (+ index 1))))))

(define vec1 '(1 2 3 4 5))
(define vec2 '(3 5 7 9))
(find-common-minimum vec1 vec2)

(define english-letters
  (map integer->char (range (char->integer #\a) (char->integer #\z) 1)))

(define (english-letter? char)
  (and (char-alphabetic? char)
       (member (char-downcase char) english-letters)))

(define (english-char? char)
  (and (char? char) (english-letter? char)))

(define (process-queue original-queue)
  (define new-queue '())

  (let loop ((remain-queue original-queue))
    (cond ((null? remain-queue)
           (display "Оригінальна черга: ")
           (display original-queue)
           (newline)
           (display "Нова черга з англійськими символами: ")
           (display new-queue)
           (newline))
          ((english-char? (car remain-queue))
           (set! new-queue (append new-queue (list (car remain-queue))))
           (loop (cdr remain-queue)))
          (else
           (loop (cdr remain-queue))))))

(define original-queue '(#\a #\b #\1 #\c #\d #\2 #\e #\F #\в))
(process-queue original-queue)
